>>> directory = "data/regtech/"

>>> from techminer2 import bibliometrix
>>> nnet = bibliometrix.clustering.coupling_network(
...     criterion="article",
...     coupling_measured_by='local_references',
...     topics_length=50,
...     metric="local_citations",
...     directory=directory,
...     method="louvain",
...     nx_k=0.5,
...     nx_iterations=10,
